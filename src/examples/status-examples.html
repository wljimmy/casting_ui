动态更新示例

// 更新状态栏项目
function updateStatus(type, text) {
    const item = document.getElementById('statusItem1');
    item.dataset.type = type;
    item.dataset.text = text;
}

// 切换闪动效果
function toggleFlash() {
    const item = document.getElementById('statusItem1');
    item.dataset.flash = item.dataset.flash === 'true' ? 'false' : 'true';
}

// 添加新的状态项
function addStatusItem() {
    const statusBar = document.getElementById('dynamicStatus');
    const newItem = document.createElement('div');
    newItem.dataset.text = '新状态';
    newItem.dataset.type = 'info';
    newItem.title = '新状态的详细说明';
    statusBar.appendChild(newItem);
}

// 随机改变颜色
function changeColor() {
    const item = document.getElementById('statusItem1');
    const colors = ['#ff6b6b', '#4ecdc4', '#45b7d1', '#96ceb4', '#ffcc5c'];
    const randomColor = colors[Math.floor(Math.random() * colors.length)];
    item.style.setProperty('--status-color', randomColor);
}
