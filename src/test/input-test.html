JavaScript API 测试

            

                
                    

                        API
                        说明
                        示例
                    

                
                
                    

                        CUI.input.get(path)
                        获取节点信息
                        CUI.input.get('loginForm.username')
                    

                    

                        CUI.input.getValue(path)
                        获取input的值
                        CUI.input.getValue('search')
                    

                    

                        CUI.input.setValue(path, value)
                        设置input的值
                        CUI.input.setValue('search', 'test')
                    

                    

                        CUI.input.getFormData(formId)
                        获取整个表单数据对象
                        CUI.input.getFormData('loginForm')
                    

                    

                        CUI.input.setFormData(formId, data)
                        设置整个表单数据
                        CUI.input.setFormData('loginForm', {username: 'test'})
                    

                    

                        CUI.input.element(path)
                        获取input的DOM元素
                        CUI.input.element('search')
                    

                    

                        CUI.input.getAll()
                        获取完整注册表
                        CUI.input.getAll()
